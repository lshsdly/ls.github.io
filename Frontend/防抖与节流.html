    # 防抖与节流

    2019/04/28

    # 含义

    • debounce 函数防抖：

    当持续触发事件时，一定时间段内没有再触发事件，事件处理函数才会执行一次，如果设定的时间到来之前，又一次触发了事件，就重新开始计时。

    • throttle 函数节流：

    当持续触发事件时，保证一定时间段内只调用一次事件处理函数。

    # 目的

    避免回调函数中的处理每次都执行

    # 防抖 & 节流区别

    • 防抖必须在一个周期内只执行一次
    • 节流是事件不停触发，间隔时间才会完成执行

    # 应用场景

    防抖：

    1. 搜索
    2. window 触发 resize - 节流也可以
    3. 防止重复提交

    节流：

    1. 鼠标不断点击触发 mousedown
    2. 监听滚动时间，如是否滑到底部自动加载更多

    # 实现及优化

    # 防抖

    <div id="demo"></div>
<script>
  var count = 1;
  var container = document.getElementById('demo');
  function getUserAction() {
    // 大量算法
    // 异步
    container.innerHTML = count++;
    //方法2 this.innerHTML = count++;
  }
  container.onmousemove = debounce(getUserAction, 1000, true);  // 要代理的函数  等待时间 1s  开始边界(首次执行)
  // 控制 getUserAction 方法在合适的时机执行
  function deboune(fn, wait, flag) {  // 代理函数
    var timeout;
    return function() { // 因为 mouseover 是绑定的函数，所以需要返回函数
      if (flag) {
        var callNow = !timeout;
        clearTimeout(timeout);
        timeout = window.setTimeout(function() {  // 只要 setTimeout 被定义，timeout 为真
          timeout = null;
        }, wait);
        if (callNow) {  // 判断是否第一次
          fn.apply(this); // 立刻执行，不能用 bind
        }
      } else {
        clearTimeout(timeout);  // 清除之前触发的 n 次定时器
        timeout = setTimeout(fn, wait);
        //方法2 timeout = window.setTimeout(fn.bind(this), wait); bind 返回新的函数，不立刻执行
      }
    }
  }
</script>

    # 节流

    // dom 同上
container.onmouseover = throttle(getUserAction ,1000);
function throttle(fn, wait) {
  var previous = 0; // 参照物
  return function() {
    var now = + new Date(); // 调用原型内的 valueOf 方法，获取当前时间戳（19700000 至今的毫秒数）
    if (now - previous > wait) {
      fn.apply(this);
      previous = now;
    }
  };
}
