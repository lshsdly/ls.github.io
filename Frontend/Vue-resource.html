            # 使用

            // main.js
import VueResource from 'vue-resource';
Vue.use(VueResource);

            Vue.use(); 告知 Vue 在其核心功能上添加一个新的插件，即拓展其核心功能。

            Firebase: 可以为你的原生应用或 web 应用提供后端服务、具体服务器存储功能、用户验证等功能。

            // $http 是 vue-resource 提供的对象，该对象上有一系列方法，每个 HTTP 请求类型都有一个对应的方法。
this.$http.post(url, data)	// post 是用来追加数据的，不是覆盖，而是追加，会确保每个数据都有一个唯一标识
	.then(res=> {	// 它允许我们提供一个函数，在异步操作完成时执行，即请求发送完成并收到相应
		console.log(res);
	}, error=> {	// then 可传递第二个函数，用于处理返回的错误
		console.log(error);
	});

// get
this.$http.get(url)
	.then(res=> {
		return res.json(); // json() 抽取出返回的 body 对象，将之转换成可以直接使用的对象
	})
	.then(data=> {	// 在这个函数内使用得到的数据
		console.log(data);
	})
// 错误写法
this.$http.get(url)
	.then(res=> {
		const data = res.json();	// 这里获取到的 data 是一个 Promise
		// 原因：res.json() 返回的是一个 promise，而非立刻返回需要的数据，因为 json() 也是一个异步操作。
	})

            # 拦截器

            Vue-resource 可以为每个请求和相应添加拦截器。

            设置 interceptors 属性并配置一组函数，每次请求都会执行这些函数； 可以给它添加一个新的拦截器，这个数组里的拦截器都会执行。

            • 拦截器结构：
              • 第一个参数 request
              • 第二个参数 next
                • 允许请求继续执行的回调函数，毕竟不能让请求都终止在这个拦截器中

            # 拦截请求

            // main.js
// 引入及应用 Vue-resource
Vue.http.interceptors.push((request, next)=> {
	console.log(request);
	if (request.method === 'POST')
		request.method = 'PUT';
	next();
});

            # 拦截响应

            // main.js
// 引入及应用 Vue-resource
Vue.http.interceptors.push((request, next)=> {
	console.log(request);
	if (request.method === 'POST')
		request.method = 'PUT';
	// 可以通过访问 next 函数、在其中传入一个新参数来查看响应，这个参数是一个函数，响应就是其参数——这是 vue-resource 提供的
	next(response=> {
		// 重写 json() 用于提取数据的动作（仅作为参考）
		response.json = ()=> {
			return {
				messages: response.body
			}
		}
	});
});

            # resource 来历

            因为它有一个附加功能，可以让我们建立自己的“资源”，就是将普通任务映射为 http 请求。

            当希望将数据存到数据库时，通常会对数据调用 save 方法，然后会自动执行 post 请求，将 data 发送到对应的 url。

            // App.vue
export default {
	data() {
		return {
			user: {},
			// 想使用 resource，首先要创建
			resource: {}
		}
	},
	// 初始化资源
	created() {
		// $resource(资源文件, )
		this.resource = this.$resource('data.json');
		// vue-resource 有一些默认的操作可用于我们的资源和数据
	},
	methods: {
		submit() {
			// 使用，向指定 Url 发送 Post 请求
			// 可传入两个参数：想要传给 Url 的参数，实际要传的数据
			this.resource.save({}, this.user);
		}
	}
}

            # 创建自定义 resource

            export default {
	data() {
		return {
			user: {},
			resource: {}
		}
	},
	created() {
		const customActions = {
			saveAlt: {
				method: 'POST',
				url: 'alternative.json'
			}
		}
		this.resource = this.$resource('data.json', {}, customActions);	// 第二个参数留空，方便将不同数据传入 URL 中
	},
	methods: {
		submit() {
			// this.resource.save({}, this.user);
			this.resource.saveAlt(this.user);
		}
	}
}

            # 理解 URL 模板

            const customActions = {
	saveAlt: {
		method: 'POST',
		url: 'alternative.json'
	},
	getData: {
		method: 'GET'
	}
}
this.resource = this.$resource('data.json', {}, customActions);
// 令 data.json 为动态：
this.resource = this.$resource('{node}/{another}.json', {}, customActions);	// data 中添加 node/another 属性
// 使用：
this.resource.getData({node: this.node, another: this.another});
